public class Package
{
[Required(ErrorMessage = “Tracking number is required.”)]
[StringLength(20, MinimumLength = 5, ErrorMessage = “Tracking number must be between 5 and 20 characters.”)]
[RegularExpression(@”\S+”, ErrorMessage = “Tracking number cannot be empty or whitespace.”)]
public string TrackingNumber { get; set; } = string.Empty;

    [Required(ErrorMessage = "Sender name is required.")]
    [RegularExpression(@"\S+", ErrorMessage = "Sender name cannot be empty or whitespace.")]
    public string Sender { get; set; } = string.Empty;

    [Required(ErrorMessage = "Recipient name is required.")]
    [RegularExpression(@"\S+", ErrorMessage = "Recipient name cannot be empty or whitespace.")]
    public string Recipient { get; set; } = string.Empty;

    [Required(ErrorMessage = "Status is required.")]
    [RegularExpression(@"\S+", ErrorMessage = "Status cannot be empty or whitespace.")]
    public string Status { get; set; } = string.Empty;
}

請幫我寫 sql指令

GPT:
USE logisticsdb;

CREATE TABLE Package (
TrackingNumber VARCHAR(20) NOT NULL,
Sender VARCHAR(255) NOT NULL,
Recipient VARCHAR(255) NOT NULL,
Status VARCHAR(255) NOT NULL,
PRIMARY KEY (TrackingNumber),
CONSTRAINT chk_TrackingNumber_Length CHECK (LENGTH(TrackingNumber) BETWEEN 5 AND 20),
CONSTRAINT chk_TrackingNumber_NotWhitespace CHECK (TrackingNumber NOT LIKE ‘% %’),
CONSTRAINT chk_Sender_NotWhitespace CHECK (Sender NOT LIKE ‘% %’),
CONSTRAINT chk_Recipient_NotWhitespace CHECK (Recipient NOT LIKE ‘% %’),
CONSTRAINT chk_Status_NotWhitespace CHECK (Status NOT LIKE ‘% %’)
);
