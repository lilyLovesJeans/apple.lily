第 1 部分：添加更多 DAO 方法

  1. 更新包裹的方法：

    • 在您的 DAO 類別中添加一個名為 updatePackage 的方法，用於更新包裹的狀態或其他屬性。
    • 方法應接收一個 Package 物件（包含更新後的所有屬性），並根據 TrackingNumber 來執行更新。
    public bool UpdatePackage(Package package)
{
    string query = "UPDATE Packages SET Sender = @Sender, Recipient = @Recipient, Status = @Status WHERE TrackingNumber = @TrackingNumber";
    using (var connection = new SqlConnection(connectionString))
    {
        using (var command = new SqlCommand(query, connection))
        {
            command.Parameters.AddWithValue("@Sender", package.Sender);
            command.Parameters.AddWithValue("@Recipient", package.Recipient);
            command.Parameters.AddWithValue("@Status", package.Status);
            command.Parameters.AddWithValue("@TrackingNumber", package.TrackingNumber);
            
            connection.Open();
            int rowsAffected = command.ExecuteNonQuery();
            return rowsAffected > 0;
        }
    }
}

  2. 刪除包裹的方法：

    • 添加一個名為 deletePackage 的方法，用於根據 TrackingNumber 刪除包裹記錄。
    public bool DeletePackage(string trackingNumber)
{
    string query = "DELETE FROM Packages WHERE TrackingNumber = @TrackingNumber";
    using (var connection = new SqlConnection(connectionString))
    {
        using (var command = new SqlCommand(query, connection))
        {
            command.Parameters.AddWithValue("@TrackingNumber", trackingNumber);
            
            connection.Open();

        }
    }
}

第 3 部分：加入異常處理

  1. 資料庫操作中的異常處理：

    • 在每個 DAO 方法中加入 try-catch 塊，以便在發生異常時能夠捕捉並處理，例如資料庫連接失敗或 SQL 語句執行失敗。
    • 可以記錄異常訊息，並根據需要返回錯誤碼或自定義的異常類型。
    public bool UpdatePackage(Package package)
{
    string query = "UPDATE Packages SET Sender = @Sender, Recipient = @Recipient, Status = @Status WHERE TrackingNumber = @TrackingNumber";
    try
    {
        using (var connection = new SqlConnection(connectionString))
        {
            using (var command = new SqlCommand(query, connection))
            {
                command.Parameters.AddWithValue("@Sender", package.Sender);
                command.Parameters.AddWithValue("@Recipient", package.Recipient);
                command.Parameters.AddWithValue("@Status", package.Status);
                command.Parameters.AddWithValue("@TrackingNumber", package.TrackingNumber);
                
                connection.Open();
                int rowsAffected = command.ExecuteNonQuery();
                return rowsAffected > 0;
            }
        }
    }
    catch (SqlException ex)
    {
        // 記錄異常訊息並處理
        Console.WriteLine("An error occurred while updating the package: " + ex.Message);
        return false;
    }
}
